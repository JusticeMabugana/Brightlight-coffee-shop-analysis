  ---TESTING
SELECT TRANSACTION_TIME
FROM "BRIGHTLIGHTS"."PUBLIC"."COFEE1";




SELECT SUM(TRANSACTION_QTY*UNIT_PRICE) AS TOTAL_REV,
FROM "BRIGHTLIGHTS"."PUBLIC"."COFEE1";


SELECT MIN(UNIT_PRICE) AS MIN, MAX(UNIT_PRICE) AS MAX
FROM "BRIGHTLIGHTS"."PUBLIC"."COFEE1";
SELECT PRODUCT_CATEGORY, UNIT_PRICE,
FROM "BRIGHTLIGHTS"."PUBLIC"."COFEE1",
ORDER BY UNIT_PRICE DESC;
SELECT *,
FROM "BRIGHTLIGHTS"."PUBLIC"."COFEE1",
ORDER BY UNIT_PRICE DESC  LIMIT 10;

SELECT DISTINCT STORE_LOCATION, 
FROM "BRIGHTLIGHTS"."PUBLIC"."COFEE1";

SELECT STORE_LOCATION, 
CASE 
WHEN TRANSACTION_TIME BETWEEN '06:00:00' AND '11:59:59' THEN 'MORNING'
WHEN TRANSACTION_TIME BETWEEN '12:00:00' AND '16:59:59' THEN 'AFTERNOON'
WHEN TRANSACTION_TIME BETWEEN '17:00:00' AND '20:59:59' THEN 'EVENING'
ELSE 'UNOPERATIONAL HOURS'
END AS TIME_CATEGORY,
 SUM(TRANSACTION_QTY*UNIT_PRICE) AS REVENUE,
FROM "BRIGHTLIGHTS"."PUBLIC"."COFEE1",
GROUP BY 
STORE_LOCATION,
TIME_CATEGORY;

SELECT TRANSACTION_DATE,
TO_CHAR(TRANSACTION_DATE,'YYYYMM') AS MONTH_ID,
FROM "BRIGHTLIGHTS"."PUBLIC"."COFEE1";

SELECT MAX(TRANSACTION_TIME), MIN (TRANSACTION_TIME),
FROM "BRIGHTLIGHTS"."PUBLIC"."COFEE1";

---FINAL PRODUCT
SELECT STORE_LOCATION, TRANSACTION_ID, TRANSACTION_DATE, TRANSACTION_TIME, UNIT_PRICE*TRANSACTION_QTY AS REVENUE,TRANSACTION_QTY,PRODUCT_CATEGORY,PRODUCT_DETAIL,PRODUCT_ID,PRODUCT_TYPE, 
SUM(TRANSACTION_QTY) AS TOTAL_QUANTITY,
TO_CHAR (TRANSACTION_DATE, 'YYYY') AS MONTH_ID,
MONTHNAME (TRANSACTION_DATE) AS MONTH_NAME,
DAYNAME (TRANSACTION_DATE) AS DAY_NAME,
CASE 
WHEN TRANSACTION_TIME BETWEEN '06:00:00' AND '11:59:59' THEN 'MORNING'
WHEN TRANSACTION_TIME BETWEEN '12:00:00' AND '16:59:59' THEN 'AFTERNOON'
WHEN TRANSACTION_TIME BETWEEN '17:00:00' AND '20:59:59' THEN 'EVENING'
ELSE 'UNOPERATIONAL HOURS'
END AS TIME_CATEGORY,
FROM "BRIGHTLIGHTS"."PUBLIC"."COFEE1",
GROUP BY 
TRANSACTION_ID,
TRANSACTION_TIME,
TRANSACTION_DATE,
REVENUE,
TRANSACTION_QTY,
TIME_CATEGORY,
MONTH_ID,
MONTH_NAME,
STORE_LOCATION,
PRODUCT_CATEGORY,
PRODUCT_DETAIL,
PRODUCT_ID,
PRODUCT_TYPE,
DAY_NAME;
